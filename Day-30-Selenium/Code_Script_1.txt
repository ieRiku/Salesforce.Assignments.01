System.setProperty("webdriver.chrome.driver", "E:\\chromedriver.exe");
				WebDriver driver = new ChromeDriver();
				driver.get("http://www.deadlinkcity.com/");
				//driver.get("https://opensource-demo.orangehrmlive.com/web/index.php/auth/login");
				Thread.sleep(5000);
				List<WebElement> allLinks=driver.findElements(By.tagName("a"));
				System.out.println("Total number of Links:"+ allLinks.size());
				for(WebElement linkElement:allLinks)
				{
					String hrefattValue=linkElement.getAttribute("href");
					if(hrefattValue==null || hrefattValue.isEmpty())
					{
						System.out.println("href attriute value is null or empty.Not possible to chech");
						continue;
					}
					URL linkURL=new URL(hrefattValue);//COnverted href value from string to url format
					HttpURLConnection conn=(HttpURLConnection)linkURL.openConnection();
					conn.connect(); //connect to server and sent request the server
					if(conn.getResponseCode()?400)
					{
						System.out.println("Broker link");
					}
					else
					{
						System.out.println("Not a Broker link");
					}
 
	}
