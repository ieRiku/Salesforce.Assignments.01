package BasePackage;
 
import java.io.IOException;

import java.time.Duration;
 
import org.openqa.selenium.WebDriver;

import org.openqa.selenium.chrome.ChromeDriver;

import org.openqa.selenium.edge.EdgeDriver;
 
import Utilities.ConfigReader;

import io.github.bonigarcia.wdm.WebDriverManager;
 
public class BaseClass {
 
	public WebDriver driver=null;

	ConfigReader con;

	public void OpenBrowser(String browser) throws InterruptedException, IOException {

		if(browser.equalsIgnoreCase("chrome")) {

		WebDriverManager.chromedriver().setup();

		driver = new ChromeDriver();

		Thread.sleep(2000);

		}

		else if(browser.equalsIgnoreCase("edge")) {

			WebDriverManager.edgedriver().setup();

			driver = new EdgeDriver();

			Thread.sleep(2000);

		}

		driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(20));

		driver.manage().window().maximize();

		con = new ConfigReader();

		String url = con.getProperty("url");

		driver.get(url);		

	}

	public void CloseBrowser() {

		driver.quit();

	}

}



package Utilities;
 
import java.io.File;

import java.io.IOException;
 
import javax.xml.parsers.DocumentBuilder;

import javax.xml.parsers.DocumentBuilderFactory;

import javax.xml.parsers.ParserConfigurationException;
 
import org.w3c.dom.Document;

import org.w3c.dom.Element;

import org.w3c.dom.Node;

import org.w3c.dom.NodeList;

import org.xml.sax.SAXException;
 
 
public class XMLData {

	public static String getData(String tag) throws ParserConfigurationException, SAXException, IOException {

		File f = new File("src/test/resources/data.xml");

		DocumentBuilderFactory df = DocumentBuilderFactory.newInstance();

		DocumentBuilder db = df.newDocumentBuilder();

		Document d = db.parse(f);

		d.getDocumentElement().normalize();

		NodeList nl = d.getElementsByTagName("user");

		Node n= nl.item(0);

		Element e = (Element)n;

		return e.getElementsByTagName(tag).item(0).getTextContent();

	}

}
