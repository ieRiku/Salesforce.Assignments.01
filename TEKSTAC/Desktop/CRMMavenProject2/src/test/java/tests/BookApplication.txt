import BasePackage.BaseClass;
import BasePackage.ExtentManager;
import Pages.HomePage;
import Pages.LoginPage;
 
@Listeners(ListenTest.class)
public class BookApplication extends BaseClass{
	LoginPage lp;
	HomePage hp;
	ExtentReports extent;
	ExtentTest test;
	Logger logger = LogManager.getLogger(BookApplication.class);
	@BeforeTest
	@Parameters("browser")
	public void LaunchBrowser(String str) throws InterruptedException {
		OpenBrowser("https://demoqa.com/login",str);
		logger.info("opened the browser");
		extent = ExtentManager.getReport();
	}
	@BeforeClass
	public void SetUpPages() {
		lp = new LoginPage(driver);
		hp = new HomePage(driver);
	}
	@AfterTest
	public void QuitBrowser() {
		CloseBrowser();
		logger.info("closed the browser");
		extent.flush();
	}
	@DataProvider(name = "dp")
	public Object[][] data() {
		return new Object[][] {{"TestUserJava","Test@123"},{"TestingUser","Test@123"}};
	}
	@Test(groups = "regression",dataProvider = "dp")
	public void LoginTest(String user, String pwd) {
		test = extent.createTest("Test case1");
		test.log(Status.INFO, "test case started");
		lp.EnterUsername(user);
		lp.EnterPassword(pwd);
		lp.ClickLogin();
		test.log(Status.INFO,"successful login");
		logger.info("test executed");
	}
	@AfterMethod
	public void LogoutTest() {
		hp.ClickLogout();
	}
	@Test(retryAnalyzer = Rerun.class)
	public void SkipTest() {
		Assert.assertEquals("xx", "xxx");
		test.log(Status.FAIL, "test case failed");
	}
}
